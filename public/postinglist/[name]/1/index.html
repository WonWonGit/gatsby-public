object

  • 키와 값 쌍으로 데이터를 저장한다
//데이터 저장 {key:value}
const hubby = {
	firstName: "Velamsson",
	lastName: "Mattias",
	age: 30,
	eyeColor: "grayBlue"
	};

//데이터 변경하는 방법
hubby["eyeColor"] = "blue";
hubby.eyeColor = "blue"

//삭제방법!
delete hubby.firstName;

//특정 키의 데이터 가져오는 방법
console.log(hubby['firstName']); //Velamsson
console.log(hubby.firstName);

//키 가져오는 방법
console.log(Object.keys(hubby)); //["firstName", "lastName", "age", "eyeColor"]

//모든 value값 가져오는 방법
console.log(Object.values(hubby)); //["welamsson", "Mattias", 30, "grayBlue"]

연산자

할당연산자

  • 오른쪽 피연산자의 값을 왼쪽 피연산자에 할당하는 것
let x=1;
let y=1;
x = x+y; //x에 x+y할당 2가됨
x = x-y; //x에 x-y할당 1이됨
x = 3;
y = 2;
x = x*y; //x에 x*y할당 6이됨
x = x/y; //x에 x/y할당 3이됨
x = x%y; //x에 x%y할당 1이됨 (나머지)
x = 3
x = x**y; //x에 x**y할당 9가됨

비교연산자

  • 피연산자들을 비교하여 논리값을 반환한다
let x = 3;
let y = 4;
x == y //false
x != y //true

let x = "3";
let y = 3;
x == y //true
x === y //false 타입까지 확인하고 싶을경우 ===를 사용해야한다

산술연산자

연산자예제
나머지연산자(%)12%5 = 2
증가연산자(++)var x=3;
var y=++x;
//y=4, x=4
var x=3
var y=x++;
//y=3, x=4
연산자의 위치에 따라 순서가 달라짐
감소연산자(—)var x=3;
var y=—x;
//y=2, x=2
var x=3
var y = x—;
//y=3, x=2
단항부정연산자(-)x가 3이면 -x는 -3
숫자화 연산자(+)+”3”은 3
+true 1
+false 0

논리연산자

  • Boolean 값에 대한 논리 AND, OR비교를 통해 참/거짓을 반환합니다.
let o1 = true && true;
let o2 = true && false;
let o3 = true || false;
let o4 = (3 == 3) && (4 == 4);
console.log(o1, o2, o3, o4);

  • 논리 NOT을 이용해서 true → false / false → true 로 출력 가능하다
let e1 = !true;
let e2 = !false;
let e3 = !(3 == 3);
let e4 = !(3 == 4);
console.log(e1, e2, e3, e4);
